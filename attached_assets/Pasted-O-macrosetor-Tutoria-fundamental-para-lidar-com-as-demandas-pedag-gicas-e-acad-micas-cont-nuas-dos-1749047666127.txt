💡 Estratégia de identificação
O macrosetor Tutoria é acionado quando há dúvidas ou interações sobre o conteúdo, entrega, acompanhamento, ou processo formativo.

Diferencia-se da Secretaria, que lida com documentos e registros; e do Suporte, que lida com acessos e falhas técnicas.

function identificarMacrosetor(texto: string): 'comercial' | 'suporte' | 'cobranca' | 'secretaria' | 'tutoria' {
  const t = texto.toLowerCase();

  const palavrasTutoria = [
    'tcc', 'estágio', 'atividades', 'atividade', 'trabalhos', 'trabalho',
    'avaliação', 'avaliações', 'matéria', 'disciplina', 'disciplinas',
    'prática', 'práticas pedagógicas', 'portfólio', 'resumo', 'prova',
    'professor', 'aula', 'orientador', 'conteúdo', 'caderno de atividades',
    'prazo de entrega', 'relatório', 'ensino', 'dúvida sobre aula'
  ];

  if (palavrasTutoria.some(p => t.includes(p))) {
    return 'tutoria';
  }

  // ... regras para secretaria, suporte, comercial, cobrança

  return 'comercial'; // fallback
}
