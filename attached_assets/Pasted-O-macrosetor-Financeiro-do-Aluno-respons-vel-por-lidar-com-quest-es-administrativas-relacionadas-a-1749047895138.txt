🔍 Dica de distinção
Esse macrosetor não trata inadimplência ativa ou negociações de dívida — que devem continuar com o setor de Cobrança.

Ele lida com situações de regularização, documentação e dúvidas financeiras de alunos adimplentes ou em processo de pagamento.

function identificarMacrosetor(texto: string): 
  'comercial' | 'suporte' | 'cobranca' | 'secretaria' | 'tutoria' | 'financeiro' {
  
  const t = texto.toLowerCase();

  const palavrasFinanceiro = [
    'nota fiscal', 'nf', 'segunda via de boleto', 'boleto', 'pagamento',
    'comprovante', 'quitei', 'quitar', 'quero quitar', 'recibo', 
    'imposto de renda', 'mensalidade paga', 'declaração de pagamento',
    'baixa no sistema', 'dados para nota fiscal', 'erro no valor', 
    'pagamento duplicado', 'paguei', 'pagamento em atraso',
    'forma de pagamento', 'data de vencimento', 'parcelamento', 
    'adiantar parcelas'
  ];

  if (palavrasFinanceiro.some(p => t.includes(p))) {
    return 'financeiro';
  }

  // ... outras regras
  return 'comercial'; // fallback
}
